1    package com.example.aklny_v30.API;
2    
3    import androidx.lifecycle.LiveData;
4    import androidx.lifecycle.MutableLiveData;
5    
6    import retrofit2.Call;
7    import retrofit2.Callback;
8    import retrofit2.Response;
9    import retrofit2.Retrofit;
10   import retrofit2.converter.gson.GsonConverterFactory;
11   
12   public class repo {
13       private ApiServices apiServices;
14       private MutableLiveData<ResponsesModel> myResponse;
15   
16       public repo(){
17           myResponse = new MutableLiveData<>();
18           apiServices = new Retrofit.Builder()
19                   .baseUrl(Constants.BASE_URL)
20                   .addConverterFactory(GsonConverterFactory.create())
21                   .build()
22                   .create(ApiServices.class);
23       }
24   
25       public void getTimeZone(){
26           apiServices.getTimeZone()
27                   .enqueue(new Callback<ResponsesModel>() {
28                       @Override
29                       public void onResponse(Call<ResponsesModel> call, Response<ResponsesModel> response) {
30                           myResponse.setValue(response.body());
31                       }
32   
33                       @Override
34                       public void onFailure(Call<ResponsesModel> call, Throwable t) {
35                           myResponse.setValue(null);
36                       }
37                   });
38       }
39   
40       public LiveData<ResponsesModel> getMyResponse() {
41           return myResponse;
42       }
43   }
