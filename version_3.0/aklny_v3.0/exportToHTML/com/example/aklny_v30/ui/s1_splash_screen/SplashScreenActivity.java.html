1    package com.example.aklny_v30.ui.s1_splash_screen;
2    
3    import androidx.appcompat.app.AppCompatActivity;
4    
5    import android.content.Intent;
6    import android.os.Bundle;
7    import android.os.Handler;
8    import android.widget.Toast;
9    
10   import com.example.aklny_v30.R;
11   import com.example.aklny_v30.ui.s2_landing_screen.LandingScreenActivity;
12   import com.example.aklny_v30.ui.s5_home_screen.Activity_HomeScreen;
13   import com.google.firebase.auth.FirebaseAuth;
14   import com.google.firebase.auth.FirebaseUser;
15   
16   public class SplashScreenActivity extends AppCompatActivity {
17   
18       FirebaseAuth firebaseAuth;
19   
20       @Override
21       protected void onCreate(Bundle savedInstanceState) {
22           super.onCreate(savedInstanceState);
23           setContentView(R.layout.activity_splash_screen);
24       }
25   
26       @Override
27       protected void onResume() {
28           super.onResume();
29           firebaseAuth = FirebaseAuth.getInstance();
30           FirebaseUser user = firebaseAuth.getCurrentUser();
31   
32           Handler handler = new Handler();
33   
34           handler.postDelayed(() -> {
35               if(user != null)
36               {
37                   // User is logged in, start home screen activity.
38                   firebaseAuth.signOut();
39                   Toast.makeText(this, "You Are Already Logged In", Toast.LENGTH_SHORT).show();
40                   startActivity(new Intent(SplashScreenActivity.this, Activity_HomeScreen.class));
41               }
42               else
43               {
44                   // User is not logged in, start landing screen activity.
45                   Toast.makeText(this, "You Are Not Logged In", Toast.LENGTH_SHORT).show();
46                   startActivity(new Intent(SplashScreenActivity.this, LandingScreenActivity.class));
47               }
48   
49               finish();
50           }, 500);
51   
52   
53       }
54   }
