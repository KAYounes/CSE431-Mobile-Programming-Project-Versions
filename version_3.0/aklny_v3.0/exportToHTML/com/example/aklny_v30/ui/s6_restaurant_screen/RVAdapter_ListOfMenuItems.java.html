20   public class RVAdapter_ListOfMenuItems extends RecyclerView.Adapter<RVAdapter_ListOfMenuItems.MenuItemViewHolder> {
21       private List<MenuItemModel> menuItems;
22       private RecyclerViewOnClickListener onClickListener;
23   
24       public RVAdapter_ListOfMenuItems(List<MenuItemModel> menuItems, RecyclerViewOnClickListener onClickListener) {
25           this.menuItems = menuItems;
26           this.onClickListener = onClickListener;
27       }
28   
29       @NonNull
30       @Override
31       public MenuItemViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
32           Context context = parent.getContext();
33           RvItemMenuItemCardBinding binder;
34           binder = RvItemMenuItemCardBinding.inflate(LayoutInflater.from(context), parent, false);
35           return new MenuItemViewHolder(binder, onClickListener);
36       }
37   
38   
39       @Override
40       public void onBindViewHolder(@NonNull MenuItemViewHolder holder, int position) {
41           MenuItemModel menuItem = menuItems.get(position);
42   
43           Picasso.get().load(menuItem.getThumbnail())
44                   .placeholder(R.drawable.icon_logo_placeholder_100dp)
45                   .error(R.drawable.icon_failed_to_load_logo_100dp)
46                   .into(holder.binder.itemThumbnail);
47           holder.binder.itemName.setText(menuItem.getName());
48           holder.binder.itemDescription.setText(menuItem.getDescription());
49           holder.binder.itemPrice.setText(Double.toString(menuItem.getPrice()));
50   
51           holder.binder.getRoot().setOnClickListener(new View.OnClickListener() {
52               @Override
53               public void onClick(View view) {
54                   Log.d("onCLick", "inner > onClick");
55                   onClickListener.onRecyclerViewClickPayload(menuItem);
56               }
57           });
58       }
59   
60       public MenuItemModel getItem(int position){
61           return menuItems.get(position);
62       }
63       @Override
64       public int getItemCount() {
65           return menuItems.size();
66       }
67

69           private RvItemMenuItemCardBinding binder;
70   
71           public MenuItemViewHolder(@NonNull RvItemMenuItemCardBinding binder, RecyclerViewOnClickListener onClickListener) {
72               super(binder.getRoot());
73               this.binder = binder;
74           }
75       }
