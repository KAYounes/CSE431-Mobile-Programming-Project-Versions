9    import android.util.Log;

12   import android.view.ViewGroup;
13   import android.widget.Toast;
14   
15   import com.example.aklny_v30.ui.transparent_activities.AuthorizationActivity;
16   import com.example.aklny_v30.databinding.FragmentSignUpWithPasswordBinding;
17   import com.example.aklny_v30.ui.s5_home_screen.Activity_HomeScreen;
18   
19   public class FragmentSignUpWithPassword extends Fragment {
20       private FragmentSignUpWithPasswordBinding binder;
21       private String helperMessage;
22       private String firstName, lastName, phoneNumber, email, password;
23   
24   
25       @Override
26       public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
27           binder = FragmentSignUpWithPasswordBinding.inflate(inflater, container, false);
28   
29           binder.btnSignUpWithPassword.setOnClickListener(new View.OnClickListener() {
30               @Override
31               public void onClick(View view) {
32                   signUp();
33               }
34           });
35   
36           return binder.getRoot();
37       }
38   
39       @Override
40       public void onDestroyView() {
41           super.onDestroyView();
42           binder = null;
43       }
44   
45       @Override
46       public void onActivityResult(int requestCode, int resultCode, @Nullable Intent data) {
47           super.onActivityResult(requestCode, resultCode, data);
48   
49           Log.d("PRINT", "onActivityResult > "
50                   + requestCode
51                   + " - " + resultCode
52                   + " - " + data.getStringExtra(AuthorizationActivity.RESULT_KEY));
53   
54           if(requestCode == AuthorizationActivity.AUTHENTICATION_REQUEST_CODE){
55               String result = data.getStringExtra(AuthorizationActivity.RESULT_KEY);
56               if(result.equals(AuthorizationActivity.AUTHENTICATION_SUCCESS)){
57                   Toast.makeText(requireActivity(), "Account Created", Toast.LENGTH_SHORT).show();
58                   startActivity(new Intent(requireActivity(), Activity_HomeScreen.class));
59                   requireActivity().finish();
60               }
61   
62               Toast.makeText(requireActivity(), data.getStringExtra(AuthorizationActivity.RESULT_KEY), Toast.LENGTH_SHORT).show();
63           }
64   
65       }
66   
67   
68   
69       private void signUp() {
70           firstName = binder.textInputEditFirstName.getText().toString();
71           lastName = binder.textInputEditLastName.getText().toString();
72           phoneNumber = binder.textInputEditPhoneNumber.getText().toString();
73           email = binder.textInputEditEmail.getText().toString();
74           password = binder.textInputEditPassword.getText().toString();
75   
76           if(validateAccountDetails())
77           {
78               Intent signUpWithPasswordIntent = new Intent(requireActivity(), AuthorizationActivity.class);
79               signUpWithPasswordIntent.putExtra("ACTION", AuthorizationActivity.SIGN_UP_WITH_PASSWORD);
80               signUpWithPasswordIntent.putExtra("FIRSTNAME", firstName);
81               signUpWithPasswordIntent.putExtra("LASTNAME", lastName);
82               signUpWithPasswordIntent.putExtra("PHONENUMBER", phoneNumber);
83               signUpWithPasswordIntent.putExtra("EMAIL", email);
84               signUpWithPasswordIntent.putExtra("PASSWORD", password);
85               startActivityForResult(signUpWithPasswordIntent, AuthorizationActivity.AUTHENTICATION_REQUEST_CODE);
86           }
87           else
88           {
89               binder.textHelpMessage.setText(helperMessage);
90           }
91       }
92   
93       private boolean validateAccountDetails(){
94           return validateName() && validatePhoneNumber() && validateEmail() && validatePassword();
95       }
96   
97       private boolean validateName() {
98   
99           if(firstName.isEmpty() || lastName.isEmpty()){
100              helperMessage = "Your name cannot be empty";
101              return false;
102          }
103  
104          if(!firstName.matches("[a-zA-Z ,.'-]{2,15}") && lastName.matches("[a-zA-Z ,.'-]{2,15}")){
105              helperMessage = "Name is invalid";
106              return  false;
107          }
108  
109          return true;
110      }
111  
112      private boolean validatePhoneNumber() {
113  
114          if(phoneNumber.isEmpty()){
115              helperMessage = "Phone number cannot be empty";
116              return false;
117          }
118  
119          if(!phoneNumber.matches("[0-9]{10}")){
120              helperMessage = "Phone number is not a valid Egyptian Number";
121              return  false;
122          }
123  
124          return true;
125      }
126  
127      private boolean validateEmail() {
128  
129          if(email.isEmpty()){
130              helperMessage = "Email cannot be empty";
131              return false;
132          }
133  
134          if(!email.matches("^[a-zA-Z0-9_!#$%&amp;'*+/=?`{|}~^-]+(?:\\.[a-zA-Z0-9_!#$%&amp;'*+/=?`{|}~^-]+)*@[a-zA-Z0-9-]+(?:\\.[a-zA-Z0-9-]+)*$")){
135              helperMessage = "Email is not a valid email";
136              return  false;
137          }
138  
139          return true;
140      }
141  
142      private boolean validatePassword() {
143  
144          if(password.isEmpty()){
145              helperMessage = "Password cannot be empty";
146              return false;
147          }
148  
149          if(!password.matches("^(?=.*[0-9])(?=.*[a-z])(?=.*[A-Z])(?=.*[@#$%^&\\-+=()]).{8,20}$")){
150              helperMessage = "Make sure your password is 8 characters long, has an uppercase and a lowercase letter, has a number, and has a symbol";
151              return  false;
152          }
153  
154          return true;
155      }
156  }
