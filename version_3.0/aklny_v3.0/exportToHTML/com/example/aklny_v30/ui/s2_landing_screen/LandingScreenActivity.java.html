77       private boolean isNetworkAvailable(){
78           try
79           {
80               ConnectivityManager connectivityManager = (ConnectivityManager) getSystemService(Context.CONNECTIVITY_SERVICE);
81               NetworkInfo networkInfo = null;
82               if(connectivityManager != null){
83                   networkInfo = connectivityManager.getActiveNetworkInfo();
84               }
85               return networkInfo != null && networkInfo.isConnected();
86           }
87           catch (NullPointerException e)
88           {
89               return false;
90           }
91       }
