140      private boolean validateMenu(){
141          if(generatedMenuKey.isEmpty()){
142              displayValidationMessage("Menu key empty, please check again.");
143              return false;
144          }
145  
146          Log.i("PRINT", "length " + generatedMenuKey.length() + " " +  "-NJwKs10ABmOCm9E51oW".length());
147          if(generatedMenuKey.length() != "-NJwKs10ABmOCm9E51oW".length()){
148              displayValidationMessage("Menu key error, please check again.");
149              return false;
150          }
151  
152          if(!menuTitle.matches(Constants.PATTERN_MENU_TITLE)) {
153              displayValidationMessage("Menu name error, please check again.");
154              return false;
155          }
156  
157          if(listOfMenuItems.size() == 0){
158              displayValidationMessage("No items were added, please check again.");
159              return false;
160          }
161  
162          return true;
163      }
164      private boolean validateMenuItem() {
165          if(!itemName.matches(Constants.PATTERN_RESTAURANT_NAME)) {
166              displayValidationMessage("Item name error, please check again.");
167              return false;
168          }
169  
170          if(!itemDescription.matches(Constants.PATTERN_RESTAURANT_DESCRIPTION)) {
171              displayValidationMessage("Description error, please check again.");
172              return false;
173          }
174  
175          if((itemPrice < 0.0) || (itemPrice > 1000.0)) {
176              displayValidationMessage("Price error, please check again.");
177              return false;
178          }
179  
180          if(itemThumbnailUrl.isEmpty()) {
181              displayValidationMessage("Thumbnail Url error, please check again.");
182              return false;
183          }
184  
185          return true;
186      }
187
