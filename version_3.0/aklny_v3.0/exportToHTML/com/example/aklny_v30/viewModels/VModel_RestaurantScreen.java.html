25           super(application);

43   
44   
45   /** Cart Interactions **/
46       public LiveData<List<CartItemModel>> getCart()
47       {
48           return cart;
49       }
50

52       {
53           cartRepo.deleteCart();
54       }
55   
56       public int getCartItemsCount()
57       {
58           int count = 0;
59           for(CartItemModel cartItem: cart.getValue())
60           {
61               count += cartItem.getQuantity();
62           }
63           return count;
64       }
65   
66       public double getCartSubtotal()
67       {
68           double subtotal = 0;
69           for(CartItemModel cartItem: cart.getValue())
70           {
71               subtotal += cartItem.getPrice() * cartItem.getQuantity();
72           }
73   
74           return subtotal;
75       }
