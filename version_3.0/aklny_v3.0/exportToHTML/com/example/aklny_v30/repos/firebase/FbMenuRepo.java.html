40       public Task<Void> addMenuModelToFB(String menuKey, MenuModel newMenu){
41           String menuItemKey;
42           List<String> listOfMenuItemKeys = new ArrayList<>();
43           HashMap<String, Object> menuRefUpdates = new HashMap<>();
44           HashMap<String, Object> menuItemsRefUpdates = new HashMap<>();
45   
46           for(MenuItemModel item: newMenu.getMenuItems()){
47               menuItemKey = menusRef.child(menuKey).child(newMenu.getTitle()).push().getKey();
48               listOfMenuItemKeys.add(menuItemKey);
49               menuItemsRefUpdates.put(menuKey + "/" + menuItemKey , item);
50               menuRefUpdates.put(menuKey + "/" + newMenu.getTitle() + "/" + menuItemKey, item);
51           }
52   
53   //        menusItemRef.updateChildren(menuItemsRefUpdates)
54           return menusRef.updateChildren(menuRefUpdates);
55       }

58       public void attachPersistentListener(String menuKey){
59           menusRef.child(menuKey).addValueEventListener(new ValueEventListener()
60           {
61               @Override
62               public void onDataChange(@NonNull DataSnapshot snapshot)
63               {
64                   try {
65                       List<MenuModel> fetchedMenus = new ArrayList<>();
66                       List<MenuItemModel> menuItems;
67                       MenuModel menu;
68                       MenuItemModel menuItem;
69   
70                       for(DataSnapshot data: snapshot.getChildren()){
71                           menu = new MenuModel();
72                           menuItems = new ArrayList<>();
73                           menu.setTitle(data.getKey());
74                           Log.d("menu", "menu title > " + menu.getTitle());
75   
76                           for(DataSnapshot item: data.getChildren()){
77                               menuItem = item.getValue(MenuItemModel.class);
78                               menuItem.setKey(item.getKey());
79                               menuItems.add(menuItem);
80                           }
81   
82                           menu.setMenuItems(menuItems);
83                           fetchedMenus.add(menu);
84                       }
85   
86                       menusLive.postValue(fetchedMenus);
87                   }
88                   catch (Exception e){
89                       Log.e("PRINT", "FbMenuRepo > attachPersistentListener > onDataChange > Exception > "
90                               + e.getLocalizedMessage());
91                   }
92               }
93   
94               @Override
95               public void onCancelled(@NonNull DatabaseError error)
96               {
97                   Log.e("PRINT", "FbMenuRepo > attachPersistentListener > onCancelled > "
98                           + error.getMessage());}
99           });
100      }
