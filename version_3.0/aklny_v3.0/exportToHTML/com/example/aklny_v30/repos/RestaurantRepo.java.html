20   public class RestaurantRepo {
21       public String recentRestaurantKey;
22       FirebaseDatabase databaseInstance = FirebaseDatabase.getInstance("https://aklny-v3-default-rtdb.firebaseio.com/");
23       public DatabaseReference restaurantsRef = databaseInstance.getReference("restaurants");
24       private MutableLiveData<List<RestaurantModel>> fetched = new MutableLiveData<>();
25   
26       public LiveData<List<RestaurantModel>> getFetched() {
27           return fetched;
28       }
29   
30       public Task<Void> addNewRestaurantToFB(RestaurantModel newRestaurant){
31           String restaurantKey = restaurantsRef.push().getKey();
32           newRestaurant.setKey(restaurantKey);
33           recentRestaurantKey = restaurantKey;
34           return restaurantsRef.child(restaurantKey).setValue(newRestaurant);
35       }
36   
37   
38       public void attachPersistentListener(){
39           restaurantsRef.addValueEventListener(new ValueEventListener() {
40               @Override
41               public void onDataChange(@NonNull DataSnapshot snapshot) {
42                   try {
43                       List<RestaurantModel> fetchedRestaurants = new ArrayList<>();
44                       String restaurantKey;
45                       RestaurantModel restaurant;
46   
47                       for(DataSnapshot data: snapshot.getChildren()){
48                           restaurantKey = data.getKey();
49                           restaurant = data.getValue(RestaurantModel.class);
50                           restaurant.setKey(restaurantKey);
51                           Log.w("PRINT", "restaurant > " + restaurant.getDeliveryFee());
52                           fetchedRestaurants.add(restaurant);
53                       }
54   
55                           fetched.postValue(fetchedRestaurants);}
56                   catch (Exception e){
57                       Log.e("PRINT", "attachPersistentListener > onDataChange > Exception > "
58                               + e.getLocalizedMessage());
59                   }
60               }
61   
62               @Override
63               public void onCancelled(@NonNull DatabaseError error) {
64                   Log.e("PRINT", "RestaurantRepo > attachPersistentListener > onCancelled > "
65                           + error.getMessage());
66               }
67           });
68       }
69   }
